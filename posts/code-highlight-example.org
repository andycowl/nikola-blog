This is the famous 'Hello world' program written in Python in colour.

#+BEGIN_SRC python
#!/usr/bin/env python
import sys

def hello(name='world'):
    greeting = "hello " + name
    print(greeting)

if __name__ == "__main__":
    hello(*sys.argv[1:])
#+END_SRC
